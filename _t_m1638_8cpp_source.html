190 uint32_t TM1638::getKeys()
191 {
192  uint32_t keys = 0;
193 
194  // Read 4 Bytes key-scan registers
195  TM1638_STB_LOW();
196  writeByte(TM1638_CMD_DATA | TM1638_DATA_READ_KEYS);
197  for (uint8_t i = 0; i < 4; i++) {
198  keys |= ((uint32_t)readByte() << (i * 8));
199  }
200  TM1638_STB_HIGH();
201 
202  return keys;

232 void TM1638::writeByte(uint8_t data)
233 {
234  for (uint8_t i = 0; i < 8; i++) {
235  TM1638_CLK_LOW();
236  TM1638_PIN_DELAY();
237  if (data & (1 << i)) {
238  TM1638_DIO_HIGH();
239  } else {
240  TM1638_DIO_LOW();
241  }
242  TM1638_CLK_HIGH();
243  TM1638_PIN_DELAY();
244  }
245 }

251 uint8_t TM1638::readByte()
252 {
253  uint8_t data = 0;
254 
255  TM1638_DIO_INPUT();
256  for (uint8_t i = 0; i < 8; i++) {
257  TM1638_CLK_LOW();
258  TM1638_PIN_DELAY();
259  if (TM1638_DIO_READ()) {
260  data |= (1 << i);
261  }
262  TM1638_CLK_HIGH();
263  TM1638_PIN_DELAY();
264  }
265  TM1638_DIO_OUTPUT();
266
