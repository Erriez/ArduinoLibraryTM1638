149 {

178 uint8_t TM1638::readByte()
179 {
180  uint8_t data = 0;
181 
182  // Set DIO pin to input
183  digitalWrite(_dioPin, LOW);
184  pinMode(_dioPin, INPUT);
185 
186  // Read 8-bit
187  for (uint8_t i = 0; i < 8; i++) {
188  digitalWrite(_clkPin, LOW);
189  if (digitalRead(_dioPin)) {
190  data |= (1 << i);
191  }
192  digitalWrite(_clkPin, HIGH);
193  }
194 
195  // Restore DIO pin to output
196  pinMode(_dioPin, OUTPUT);
197 
198  return data;
199 }
