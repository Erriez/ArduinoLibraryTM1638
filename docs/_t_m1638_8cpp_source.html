188 uint32_t TM1638::getKeys()
189 {
190  uint32_t keys = 0;
191 
192  // Read 4 Bytes key-scan registers
193  TM1638_STB_LOW();
194  writeByte(TM1638_CMD_DATA | TM1638_DATA_READ_KEYS);
195  for (uint8_t i = 0; i < 4; i++) {
196  keys |= ((uint32_t)readByte() << (i * 8));
197  }
198  TM1638_STB_HIGH();
199 
200  return keys;
201 }

230 void TM1638::writeByte(uint8_t data)
231 {
232  for (uint8_t i = 0; i < 8; i++) {
233  TM1638_CLK_LOW();
234  TM1638_PIN_DELAY();
235  if (data & (1 << i)) {
236  TM1638_DIO_HIGH();
237  } else {
238  TM1638_DIO_LOW();
239  }
240  TM1638_CLK_HIGH();
241  TM1638_PIN_DELAY();
242  }
243 }

249 uint8_t TM1638::readByte()
250 {
251  uint8_t data = 0;
252 
253  TM1638_DIO_INPUT();
254  for (uint8_t i = 0; i < 8; i++) {
255  TM1638_CLK_LOW();
256  TM1638_PIN_DELAY();
257  if (TM1638_DIO_READ()) {
258  data |= (1 << i);
259  }
260  TM1638_CLK_HIGH();
261  TM1638_PIN_DELAY();
262  }
263  TM1638_DIO_OUTPUT();
264 
265  return data;
